POST /users Register a new user 
{
  "username": "new_user",
  "password": "Password123!",
  "email": "newuser@example.com",
  "birthday": "1990-01-01"
}
 
{
  "message": "User successfully created",
  "user": { "id": "uuid", "username": "new_user", ... }
}
PUT /users/:username Update an existing user's information 
{
  "username": "updated_user",
  "password": "NewPassword123",
  "email": "updated_email@example.com",
  "birthday": "1990-01-01"
}
 
{
  "message": "User info successfully updated",
  "user": { "username": "updated_user", ... }
}
POST /users/:username/myfavorites Add a movie to a user's favorite list 
{
  "movieId": "uuid_of_movie"
}
 
{
  "message": "Movie 'Inception' has been added to favorites",
  "myFavorites": [ { "id": "uuid_of_movie", "title": "Inception", ... } ]
}
GET /users/:username/myfavorites Retrieve a user's favorite movies list N/A 
{
  "myFavorites": [ { "id": "uuid_of_movie", "title": "Inception", ... } ]
}
DELETE /users/:username/myfavorites Remove a movie from a user's favorite list 
{
  "movieId": "uuid_of_movie"
}
 
{
